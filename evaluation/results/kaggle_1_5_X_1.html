Our Solution


SELECT
    EXTRACT(year FROM trip_start_timestamp),
    count(unique_key) 
FROM
    taxi_trips 
GROUP BY
    EXTRACT(year FROM trip_start_timestamp) 
ORDER BY
    EXTRACT(year FROM trip_start_timestamp) ASC
